} } function doGet(e) { return ContentService.createTextOutput("Smart TimeTable API Ver 13.44 is running."); } // ========================================== // コア・アルゴリズム // ========================================== function generateSchedule(state, currentSchedule, currentUnassigned) { const startTime = Date.now(); // ★ ブラウザのタイムアウトを防ぐため、24秒で処理を強制的に切り上げる const TIME_LIMIT = 24000; const DAYS = ['月', '火', '水', '木', '金']; const PERIODS = [1, 2, 3, 4, 5, 6]; const teacherMap = {}; if (state.teachers) state.teachers.forEach(t => teacherMap[t.id] = t.name); } // ------------------------------------------ // ハード制約チェック（★診断カウント付き） // ------------------------------------------ function canPlaceHard(lesson, d, p, schedule, state, diagnostics) { for (let target of lesson.targets) { const grade = target.split('-')[0]; } // ------------------------------------------ // ソフト制約（配置の良し悪し）のペナルティ計算 // ------------------------------------------ function evaluatePlacementPenalty(lesson, d, p, schedule, state, rules, hrMap) { let penalty = 0; lesson.teacherIds.forEach(tid => { if (schedule[d][pairP] && schedule[d][pairP].some(x => x.teacherIds.includes(tid) && !x.targets.includes(c))) penalty += 100; }); } } return penalty; } // ========================================== // スプレッドシート出力（3軸マルチシート作成） // ========================================== function exportToSpreadsheet(schedule, data) { try { const ss = SpreadsheetApp.getActiveSpreadsheet();
